148  std::string classify(const std::string &local_target,
149  const int target_pos) const {
150  int d = right - left - 1;
151  std::string s;
152  try {
153  if (d == 1 and seq.length() == 0) {
154  // deletion of a single nucleotide
155  int i = left + 1 - target_pos;
156  s = local_target.substr(left + 1 - target_pos, 1) + '-';
157  } else if (d == 0 and seq.length() == 1) {
158  // insertion of a single nucleotide
159  s = '-' + seq;
160  } else if (d == 1 and seq.length() == 1) {
161  // single-nucleotide mismatch
162  if (seq == "N"){
163  s = "N_match";
164  } else {
165  s = local_target[left + 1 - target_pos] + seq;
166  }
167  } else if (d > 1 and seq.length() == 0) {
168  s = "multinuc_deletion";
169  } else if (d == 0 and seq.length() > 1) {
170  s = "multinuc_insertion";
171  } else if (d == seq.length()) {
172  s = "multinuc_mismatch";
173  } else if (seq.length() < d) {
174  s = "complex_deletion";
175  } else if (seq.length() > d) {
176  s = "complex_insertion";
177  } else {
178  throw std::runtime_error("Error: Unknown error. Unable to classify mutation.");
179  }
180  } catch (std::out_of_range &err) {
181  throw std::runtime_error(
182  "Error: Unable to classify mutation. Mutation location falls outside local target sequence.");
183  }
184  return s;
185  }

452  std::vector<TaggedMutation> collapseMutations(const std::vector<TaggedMutation> &mutations,
453  const int max_internal_match,
454  const std::string &local_target_seq,
455  const std::string &local_target_qual,
456  const int left_target_pos) {
457  std::vector<TaggedMutation> collapsed_mutations;
458  std::vector<TaggedMutation> unmerged_mutations; // temp storage for ambiguous mutations (if any),
459  // to avoid merging with other mutations
460  if (mutations.size() > 0) {
461  for (std::vector<TaggedMutation>::const_iterator it = mutations.begin();
462  it != mutations.end();
463  ++it) {
464  TaggedMutation m(*it);
465  // if a mutation is not really a mutation (N match),
466  // it should not be merged with other mutations
467  if (m.tag=="N_match") {
468  unmerged_mutations.push_back(m);
469  } else if (collapsed_mutations.size() > 0 and
470  collapsed_mutations.back().right - 1 == m.left) {
471  // adjacent mutation
472  collapsed_mutations.back().right = m.right;
473  collapsed_mutations.back().seq += m.seq;
474  collapsed_mutations.back().qual += m.qual;
475  collapsed_mutations.back().tag = "";
476  if (m.ambig) {
477  collapsed_mutations.back().ambig = true;
478  }
479  } else if (collapsed_mutations.size() > 0 and
480  m.left - (collapsed_mutations.back().right - 1) <= max_internal_match){
481  // mutation within allowed distance for adjacency
482  collapsed_mutations.back().seq += local_target_seq.substr(collapsed_mutations.back().right-left_target_pos,
483  m.left-collapsed_mutations.back().right+1) ;
484  // ^ pick up nucs between mutations
485  collapsed_mutations.back().qual += local_target_qual.substr(collapsed_mutations.back().right-left_target_pos,
486  m.left-collapsed_mutations.back().right+1) ;
487  collapsed_mutations.back().right = m.right;
488  collapsed_mutations.back().seq += m.seq;
489  collapsed_mutations.back().qual += m.qual;
490  collapsed_mutations.back().tag = "";
491  if (m.ambig) {
492  collapsed_mutations.back().ambig = true;
493  }
494  } else {
495  // non-adjacent mutation
496  collapsed_mutations.push_back(m);
497  }
498  }
499  // iteratively strip any matches from mutation ends (these can appear as a result of
500  // shifted indels). For left-aligned ambig gaps, this only affects the 5-prime end of
501  // a mutation, so shouldn't affect reactivity profiles. Adjusting it anyway for
502  // consistency between aligners.
503  if (collapsed_mutations.size() > 0){
504  for (std::vector<TaggedMutation>::iterator it = collapsed_mutations.begin();
505  it != collapsed_mutations.end();
506  ++it){
507  // from left end
508  int new_left = (*it).left;
509  std::string new_seq = (*it).seq;
510  std::string new_qual = (*it).qual;
511  try{
512  for (int i = 0; i < (*it).seq.length(); ++i){
513  std::string c = (*it).seq.substr(i,1);
514  std::string q = (*it).qual.substr(i,1);
515  if ((*it).left + 1 + i >= (*it).right) { break; } // don't go outside of region mutation spans
516  int p = (*it).left + 1 + i - left_target_pos;
517  if (p<0) { break; }
518  std::string r = local_target_seq.substr(p, 1);
519  if (c==r){
520  ++new_left;
521  new_seq = new_seq.substr(1);
522  new_qual = new_qual.substr(1);
523  } else {
524  break;
525  }
526  }
527  }
528  catch( std::out_of_range& exception){
529  break;
530  }
531  (*it).left = new_left;
532  (*it).seq = new_seq;
533  (*it).qual = new_qual;
534 
535  // from right end
536  int new_right = (*it).right;
537  new_seq = (*it).seq;
538  new_qual = (*it).qual;
539  try{
540  for (int i = (*it).seq.length()-1; i >= 0; --i){
541  std::string c = (*it).seq.substr(i,1);
542  std::string q = (*it).qual.substr(i,1);
543  int d = (*it).seq.length() - i;
544  if ((*it).right - d <= (*it).left) { break; } // don't go outside of region mutation spans
545  int p = (*it).right - d - left_target_pos;
546  if (p<0){ break; }
547  std::string r = local_target_seq.substr(p, 1);
548  if (c==r){
549  --new_right;
550  new_seq = new_seq.substr(0,i);
551  new_qual = new_qual.substr(0,i);
552  } else {
553  break;
554  }
555  }
556  }
557  catch( std::out_of_range& exception){
558  break;
559  }
560  (*it).right = new_right;
561  (*it).seq = new_seq;
562  (*it).qual = new_qual;
563  }
564  }
565 
566  // put unmerged (mismatched Ns) mutations back into list
567  collapsed_mutations.insert(collapsed_mutations.end(),
568  unmerged_mutations.begin(),
569  unmerged_mutations.end());
570  // make sure unmerged are back in order
571  std::sort(collapsed_mutations.begin(), collapsed_mutations.end());
572  }
573  return collapsed_mutations;
574  }

713  bad_basecall = true;
714  }
715  } catch (std::out_of_range &exception) { }
716  }
717  }
718  if (not bad_basecall){
719  local_effective_depth[i] = 1;
720  }
721  }
722 
723  // in second pass, update depths for only nucs within mutations,
724  // and filter mutations by q-scores (and mutation_type, if given)
725  //
726  // mutation_type possible values: mismatch gap insert gap_multi insert_multi complex
727  for (int i=0; i<mutations.size(); ++i) {
728  TaggedMutation m(mutations[i]);
729  bool bad_mutation = false;
730 
731  // limit to specific mutation type if given
732  // FIXME: this relies on the template class being TaggedMutation
733  std::vector<std::string> mismatch_tags = {"AT", "AG", "AC",
734  "TA", "TG", "TC",
735  "GA", "GT", "GC",
736  "CA", "CT", "CG",
737  "multinuc_mismatch"};
738  std::vector<std::string> insert_tags = {"-A", "-T", "-G", "-C", "-N"};
739  std::vector<std::string> gap_tags = {"A-", "T-", "G-", "C-"};
740 
741  if (mutation_type.length() > 0) {
742  if (mutation_type == "mismatch") {
743  if (std::find(mismatch_tags.begin(), mismatch_tags.end(), m.tag) == mismatch_tags.end()) {
744  bad_mutation = true;
745  }
746  } else if (mutation_type == "insert") {
747  if (std::find(insert_tags.begin(), insert_tags.end(), m.tag) == insert_tags.end()) {
748  bad_mutation = true;
749  }
750  } else if (mutation_type == "insert_multi") {
751  if (m.tag != "multinuc_insertion") {
752  bad_mutation = true;
753  }
754  } else if (mutation_type == "gap") {
755  if (std::find(gap_tags.begin(), gap_tags.end(), m.tag) == gap_tags.end()) {
756  bad_mutation = true;
757  }
758  } else if (mutation_type == "gap_multi") {
759  if (m.tag != "multinuc_deletion") {
760  bad_mutation = true;
761  }
762  } else if (mutation_type == "complex") {
763  if (m.tag != "complex_deletion" && m.tag != "complex_insertion") {
764  bad_mutation = true;
765  }
766  }
767  }
768 
769  // look at basecalls within mutation
770  if (not bad_mutation) {
771  for (auto c : m.qual){
772  if (c-33 < min_qual){
773  bad_mutation = true;
774  break;
775  }
776  }
777  }
778  // look at neighbor on left
779  if (not bad_mutation){
780  try {
781  int k = right_mut_indices.at(m.left + 1 - left_target_pos);
782  if ( k != -1 and k != i){
783  TaggedMutation neighbor(mutations[k]);
784  if (neighbor.seq.length() > 0){
785  // look at right-most basecall in mutation
786  if (neighbor.qual.back()-33 < min_qual){
787  bad_mutation = true;
788  }
789  } else {
790  // look at basecall on other side of gap
791  int n = neighbor.left - left_target_pos;
792  if (local_target_qual.at(n)-33 < min_qual){
793  bad_mutation = true;
794  }
795  }
796  } else {
797  // no mutation on left, just look at basecall
798  int n = m.left-left_target_pos;
799  if (local_target_qual.at(n)-33 < min_qual){
800  bad_mutation = true;
801  }
802  }
803  } catch (std::out_of_range &exception) { }
804  }
805  // look at neighbor on right
806  if (not bad_mutation){
807  try{
808  int k = left_mut_indices.at(m.right - 1 - left_target_pos);
809  if (k != -1 and k != i){
810  TaggedMutation neighbor(mutations[k]);
811  if (neighbor.seq.length() > 0){
812  // look at left-most basecall in mutation
813  if (neighbor.qual.front()-33 < min_qual){
814  bad_mutation = true;
815  }
816  } else {
817  // look at basecall on other side of gap
818  int n = neighbor.right - left_target_pos;
819  if (local_target_qual.at(n)-33 < min_qual){
820  bad_mutation = true;
821  }
822  }
823  } else {
824  // no mutation on right, just look at basecall
825  int n = m.right-left_target_pos;
826  if (local_target_qual.at(n)-33 < min_qual){
827  bad_mutation = true;
828  }
829  }
830  } catch (std::out_of_range &exception) { }
831  }
832 
833  if (bad_mutation){
834  // exclude entire covered region from depths
835  for (int n = m.left + 1 - left_target_pos;
836  n < m.right - left_target_pos;
837  ++n) {
838  try {
839  local_effective_depth.at(n) = 0;
840  } catch (std::out_of_range &exception) { }
841  }
842  excluded_mutations.push_back(TaggedMutation(m));
843  } else {
844  if (variant_mode){
845  // include entire covered region (even in gap positions)
846  for (int n = m.left + 1 - left_target_pos;
847  n <= m.right - 1 - left_target_pos;
848  ++n) {
849  try {
850  local_effective_depth.at(n) = 1;
851  } catch (std::out_of_range &exception) { }
852  }
853  } else {
854  // exclude covered region, include inferred adduct site
855  for (int n = m.left + 1 - left_target_pos;
856  n < m.right - 1 - left_target_pos;
857  ++n) {
858  try {
859  local_effective_depth.at(n) = 0;
860  } catch (std::out_of_range &exception) { }
861  }
862  try {
863  local_effective_depth.at(m.right-1-left_target_pos) = 1;
864  } catch (std::out_of_range &exception) { }
865  }
866  included_mutations.push_back(TaggedMutation(m));
867  }
868  }
869 
870  // also mark mutation locations in simplified array
871  // (unused by downstream shapemapper components, but simpler to parse
872  // for single-molecule applications)
873  std::vector<bool> local_effective_count(len, 0);
874  for (auto &m:included_mutations) {
875  try {
876  local_effective_count.at(m.right - 1 - left_target_pos) = 1;
877  } catch (std::out_of_range &exception) { }
878  }
879 
880  return boost::make_tuple(local_effective_depth,
881  local_effective_count,
882  included_mutations,
883  excluded_mutations);
884  };
